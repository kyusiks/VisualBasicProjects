SELECT COLUMN_NAME          
     , MAX (COMMENTS)       
  FROM DBA_COL_COMMENTS     
 WHERE COMMENTS IS NOT NULL 
 GROUP BY COLUMN_NAME       
 
SELECT TABLE_NAME           
     , MAX (COMMENTS)       
  FROM DBA_TAB_COMMENTS     
 WHERE COMMENTS IS NOT NULL 
   AND TABLE_TYPE = 'TABLE' 
 GROUP BY TABLE_NAME
